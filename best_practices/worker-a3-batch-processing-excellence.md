```bash
# バッチサイズ: 10ファイル並行処理
BATCH_SIZE=10

# 並行削除関数
delete_batch() {
    local files=("$@")
    for file in "${files[@]}"; do
        if [ -f "$file" ]; then
            rm -f "$file" && echo "  ✓ 削除: $file"
        fi
    done
}
```
